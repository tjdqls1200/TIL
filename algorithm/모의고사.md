# 모의고사 (Programmers, Level 1)

int[] answers 배열 개수만큼의 문제 중 가장 높은 점수를 받은 학생을 구하기.

각 학생은 일정 패턴으로 번호를 찍음.

## 1. 풀이 코드

```
import java.util.ArrayList;

class Solution {
    public int[] solution(int[] answers) {
        ArrayList<Integer> temp = new ArrayList<Integer>();
        int[] student1 = {1, 2, 3, 4, 5};
        int[] student2 = {2, 1, 2, 3, 2, 4, 2, 5};
        int[] student3 = {3, 3, 1, 1, 2, 2, 4, 4, 5, 5};
        int[] grade = {0, 0, 0};

        for (int i=0; i < answers.length; i++) {
            if(answers[i] == student1[i%5]) {
                grade[0]++;
            }
            if(answers[i] == student2[i%8]) {
                grade[1]++;
            }
            if(answers[i] == student3[i%10]) {
                grade[2]++;
            }
        }
        if (grade[0] >= grade[1] && grade[0] >= grade[2]) {
            temp.add(1);
        }
        if (grade[1] >= grade[0] && grade[1] >= grade[2]) {
            temp.add(2);
        }
        if (grade[2] >= grade[0] && grade[2] >= grade[1]) {
            temp.add(3);
        }
        int[] answer = new int[temp.size()];
        for(int i = 0; i < answer.length; i++) {
            answer[i] = temp.get(i).intValue();
        }
        return answer;
    }
}
```



## 2. 참고 풀이 코드

```
import java.util.ArrayList;
class Solution {
    public int[] solution(int[] answer) {
        int[] a = {1, 2, 3, 4, 5};
        int[] b = {2, 1, 2, 3, 2, 4, 2, 5};
        int[] c = {3, 3, 1, 1, 2, 2, 4, 4, 5, 5};
        int[] score = new int[3];
        for(int i=0; i<answer.length; i++) {
            if(answer[i] == a[i%a.length]) {score[0]++;}
            if(answer[i] == b[i%b.length]) {score[1]++;}
            if(answer[i] == c[i%c.length]) {score[2]++;}
        }
        int maxScore = Math.max(score[0], Math.max(score[1], score[2]));
        ArrayList<Integer> list = new ArrayList<>();
        if(maxScore == score[0]) {list.add(1);}
        if(maxScore == score[1]) {list.add(2);}
        if(maxScore == score[2]) {list.add(3);}
        return list.stream().mapToInt(i->i.intValue()).toArray();
    }
}
```
